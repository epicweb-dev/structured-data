ğŸ’° Return tuple pattern:

```ts
function getMinMax(nums: Array<number>): [number, number] {
	let min = nums[0]
	let max = nums[0]
	for (const num of nums) {
		if (num < min) min = num
		if (num > max) max = num
	}
	return [min, max]
}
```
